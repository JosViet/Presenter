\subsection{Bình phương của một tổng, một hiệu}
\subsubsection{Kiến thức trọng tâm}
\begin{tomtat}
	Với hai biểu thức tùy ý $A$ và $B$, ta có
	\begin{itemize}
		\item $\left({A}+{B}\right)^2={A}^2+2{A}{B}+{B}^2$.
		\item $\left({A}-{B}\right)^2={A}^2-2{A}{B}+{B}^2$.
	\end{itemize}
\end{tomtat}

\begin{vd}%[Dự án EX-9-Đề Cương Toán 9]%[Viet Hoang Pham]%[8D1H3-1]
	Viết các biểu thức sau thành đa thức
	\begin{multicols}{3}
		\begin{enumerate}
			\item $(x+3)^2$;
			\item $(2x-3y)^2$;
			\item $(x^2-4y)^2$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $(x+2)^2=x^2+2\cdot x\cdot 2+2^2=x^2+4x+4$.
			\item $(2x-3y)^2=(2x)^2-2\cdot 2x\cdot 3y +(3y)^2=4x^2-12xy+9y^2$.
			\item $(a^2-5b)^2=\left(a^2\right)^2-2\cdot a^2\cdot 5b +(5b)^2=a^4-10a^2b+25b^2$.
		\end{enumerate}
	}
\end{vd}

\begin{tomtat}
	Với hai biểu thức tùy ý $A$ và $B$, ta có
	$${A}^2-{B}^2=({A}+{B})({A}-{B}). $$
\end{tomtat}

\subsection{Lập phương của một tổng, một hiệu}
\subsubsection{Kiến thức trọng tâm}
\begin{tomtat}
	Với hai biểu thức tùy ý $A$ và $B$, ta có
	\begin{itemize}
		\item $({A}+{B})^3={A}^3+3{A}^2{B}+3{A}{B}^2+{B}^3$.
		\item $({A}-{B})^3={A}^3-3{A}^2{B}+3{A}{B}^2-{B}^3$.
	\end{itemize}
\end{tomtat}

\subsection{Tổng và hiệu của hai lập phương}
\subsubsection{Kiến thức trọng tâm}
\begin{tomtat}
	Với hai biểu thức tùy ý $A$ và $B$, ta có
	\begin{itemize}
		\item ${A}^3+{B}^3=({A}+{B})\left({A}^2-{A}{B}+{B}^2\right)$.
		\item ${A}^3-{B}^3=({A}-{B})\left({A}^2+{A}{B}+{B}^2\right)$.
	\end{itemize}
\end{tomtat}

			\item $y^3-8$;
			\item $125x^3+27y^3$;
			\item $8x^3-125$;
			\item $8x^3-y^3$;
			\item $27x^3+y^3$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $8y^3+1=(2y)^3+1^3=(2y+1)\left[(2y)^2-2y\cdot 1+1^2\right]=(2y+1)\left(4y^2-2y+1\right)$;
			\item $y^3-8=y^3-2^3=(y-2)\left(y^2+2y+2^2\right)=(y-2)\left(y^2+2y+4\right)$;
			\item $125x^3+27y^3=(5x)^3+(3y)^3=(5x+3y)\left((5x)^2-5x\cdot 3y +(3y)^2\right)=(5x+3y)\left(25x^2-15xy +9y^2\right)$;
			\item $8x^3-125=(2x)^3-5^3=(2x-5)\left((2x)^2+2x\cdot 5+5^2\right)=(2x-5)\left(4x^2+10x+25\right)$;
			\item $8x^3-y^3=(2x)^3-y^3=(2x-y)\left((2x)^2+2xy+y^2\right)=(2x-y)\left(4x^2+2xy+y^2\right)$;
			\item $27x^3+y^3=(3x)^3+y^3=(3x+y)\left((3x)^2-3xy+y^2\right)=(3x+y)\left(9x^2-3xy+y^2\right)$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Dự án EX-9-Đề Cương Toán 9]%[Viet Hoang Pham]%[8D1V3-4]
	Rút gọn các biểu thức sau
		\begin{enumerate}
			\begin{multicols}{2}
			\item $(x-3y)^2-(x+3y)^2$;
			\item $(2x+4y)^2+(4x-3y)^2$;
			\item $(x-2)\left(x^2+2x+4\right)-x^3+9$;
			\item $(x-2y)^3+(x+2y)^3$;
			\item $\left[(3x+2y)(3x-2y)\right]^3$;
			\item $(3x+y)\left(9x^2-3xy+y^2\right)-(3x-y)\left(9x^2+3xy+y^2\right)$.
			\end{multicols}
		\end{enumerate}
	
	\loigiai{
		\begin{enumerate}
			\item $(x-3y)^2-(x+3y)^2=\left[(x-3y)+(x+3y)\right]\left[(x-3y)-(x+3y)\right]=2x\cdot (-6y)=-12xy$;
			\item $(2x+4y)^2+(4x-3y)^2=(2x)^2+2\cdot 2x\cdot 4y+(4y)^2+(4x)^2-2\cdot 4x\cdot 3y+(3y)^2=20x^2-8xy+25y^2$;
			\item $(x-2)\left(x^2+2x+4\right)-x^3+9=(x-2)\left(x^2+2x+2^2\right)-x^3+9=x^3-2^3-x^3+9=1$;
			\item $(x-2y)^3+(x+2y)^3=[(x-2y)+(x+2y)]\left[(x-2y)^2-(x-2y)(x+2y)+(x+2y)^2\right]=2x\left(x^2+12y^2\right)$;
			\item $(3x+2y)(3x-2y)=(3x)^2-(2y)^2=9x^2-4y^2$;
			\item $(3x+y)\left(9x^2-3xy+y^2\right)-(3x-y)\left(9x^2+3xy+y^2\right)$\\$=(3x+y)\left[(3x)^2-3x\cdot y+y^2\right]-(3x-y)\left[(3x)^2+3x\cdot y+y^2\right]=\left[(3x)^3+y^3\right]-\left[(3x)^3-y^3\right]=2y^3$.
		\end{enumerate}
	}
\end{bt}

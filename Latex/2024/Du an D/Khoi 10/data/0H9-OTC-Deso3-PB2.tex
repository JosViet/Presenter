			&\Leftrightarrow& \heva{&a-1+3(b-1)=0\\ &(a-1)^2+(b-1)^2=10} \\

		\end{eqnarray*}

		Vậy $T=3a+4b=3\cdot (-2)+4\cdot 2=2$.
	}

\begin{ex}%[0D7H1-2]
	Tìm giá trị nguyên dương nhỏ nhất của tham số $m$ để bất phương trình $x^2+6x+m+7\le 0$ vô nghiệm.
	\shortans{$3$}
	\loigiai{
		Ta có $x^2+6x+m+7\le 0$ vô nghiệm khi \[x^2+6x+m+7> 0,\forall x\in \mathbb{R}\Rightarrow \heva{&{a > 0} \\&{\Delta'< 0}&}\Rightarrow \heva{&1> 0\text{ (luôn đúng)}\\&{3^2-(m+7) < 0}}\Rightarrow m > 2.\]
		Với $m > 2$ thì bất phương trình $x^2+6x+m+7\le 0$ vô nghiệm.\\
		Vậy giá trị nguyên dương nhỏ nhất của tham số $m$ thỏa mãn là $m=3$.
	}
\end{ex}
\begin{ex}%[0D7H2-6]
	Có bao nhiêu giá trị nguyên của tham số $m$ thuộc đoạn $\left[-10;5\right]$ để phương trình $x^2-(m+1)x+3m-5=0$ có hai nghiệm phân biệt?
	\shortans{$13$}
	\loigiai{
		Ta có $\Delta=[-(m+1)]^2-4\cdot 1\cdot (3m-5)=m^2-10m+21$.\\
		Để phương trình $x^2-(m+1)x+3m-5=0$ có hai nghiệm phân biệt thì \[\Delta > 0\text{ hay } m^2-10m+21> 0.\]
		Tam thức bậc hai $m^2-10m+21$ có $a=1> 0$ và có hai nghiệm $m_1=3$, $m_2=7$.\\
		Do đó, $m^2-10m+21> 0$ khi $m\in (-\infty;3)\cup (7;+\infty)$.\\
		Mà $m\in \mathbb{Z}$ và $m\in \left[-10;5\right]$ nên có $13$ giá trị nguyên của tham số $m$ thỏa mãn.
	}
\end{ex}
\begin{ex}%[0D7V2-7]

		Vì hai ô tô xuất phát cùng một lúc nên thời gian ô tô đi từ vị trí $B$ đến địa điểm $C$ cũng là $x$ giờ.\\
		Do đó, quảng đường $AD$ và $BC$ lần lượt là $55x$ km và $45x$ km.\\
		Suy ra khoảng cách từ vị trí $A$ và vị trí $B$ đến thành phố $T$ lần lượt là $55 x+14$ (km)
		và $45 x+6$ (km).\\

		Giải phương trình này và kết hợp với điều kiện ${x>0}$, ta nhận ${x=\dfrac{6}{5}}$.\\

\begin{ex}%[0D7H1-2]
	Có bao nhiêu giá trị nguyên của tham số $m$ để biểu thức $f(x)=(m-1)x^{2}-2(m-1)x-4$ luôn không dương với mọi $x$ thuộc $\mathbb{R}$?
	\loigiai{
		Ta phải tìm $m$ sao cho $f(x)=(m-1)x^{2}-2(m-1)x-4\leq 4,\forall x\in \mathbb{R}$.
		\begin{itemize}
			\item Trường hợp 1: $m-1=0\Leftrightarrow m=1$. Khi đó $f(x)=-4<0,\forall x\in \mathbb{R}$.
			\item Trường hợp 2: $m-1\neq 0\Leftrightarrow m\neq 1$. \\
			Khi đó $f(x)\leq 0, \forall x\in \mathbb{R}$ $\Leftrightarrow\heva{&m-1<0 \\ &\Delta '=(m-1)^{2}+4(m-1)\leq 0.}$ \\
			Từ đó, suy ra $\heva{& m-1<0 \\ &\Delta '=(m-1)(m+3)\leq 0}$
			$\Leftrightarrow\heva{&m-1<0 \\ &m+3\geq 0}\Leftrightarrow -3\leq m<1$.
		\end{itemize}
		Kết hợp hai trường hợp ta suy ra giá trị $m$ cần tìm là $-3\leq m\leq 1$. \\
		Vậy có 5 giá trị nguyên của $m$.
	}
\end{ex}
\begin{ex}%[0D7V2-7]
	Một chú thỏ đen chạy đuổi theo một chú thỏ trắng ở vị trí cách nó $100$~(m). Biết rằng, quãng đường chú thỏ đen chạy được biểu thị bởi công thức $s(t)=8t+5t^2$~(m), trong đó $t$ (giây) là thời gian tính từ thời điểm chú thỏ đen bắt đầu chạy, và chú thỏ trắng chạy với vận tốc không đổi là $3$~(m/s). Hỏi bắt đầu từ giá trị nguyên $t$ nào thì chú thỏ đen chạy trước chú thỏ trắng?
	
	\loigiai{
		Giả sử vị trí ban đầu của chú thỏ đen là $s=0$~(m) và thời điểm ban đầu là $t=0$ (giây).\\
		Quãng đường của chú thỏ trắng chạy được tại thời điểm $t$ là $f(t)=100+3t$~(m).\\
		Để chú thỏ đen chạy trước chú thỏ trắng thì \[s(t) > f(t)\text{ hay }8t+5t^2 > 100+3t\Leftrightarrow 5t^2+5t-100> 0\Rightarrow t\in (4;+\infty) (\text{vì }t>0).\]
		Vậy bắt đầu giây thứ $5$ thì chú thỏ đen chạy trước chú thỏ trắng.
	}
\end{ex}
\begin{ex}%[0D7V3-6]
	Hằng ngày, bạn Hùng đạp xe đến trường, đi qua một đoạn đường thẳng (lề đường chính). Bạn Minh ở trong một con hẻm cách lề đường một đoạn $50$ mét, và không thể nhìn thấy đường chính. Để cùng đi học mà không ai phải chờ ai, hai bạn hẹn sẽ gặp nhau tại một điểm $C$ nào đó trên lề đường.\\	
	Biết rằng
	\begin{itemize}
		\item Vận tốc xe đạp của Hùng là $15$ km/h.
		\item Vận tốc đi bộ của Minh là $5$ km/h.
		\item Mỗi ngày, Hùng đều đi qua một điểm $B$ trên lề đường, mà điểm $B$ cách vị trí của Minh (tính theo đường thẳng) một đoạn đúng $200$ mét.
		\item Khi Hùng đến điểm $B$, cậu gọi điện báo cho Minh bắt đầu đi ra lề đường để hai bạn gặp nhau.
	\end{itemize}
Hai bạn sẽ gặp nhau tại điểm $C$ nào trên lề đường sao cho không ai phải đợi người kia?(làm tròn kết quả đến hàng phần muời)

	\begin{center}
		\begin{tikzpicture}[thick,scale=1.5]
			\path (0,0) coordinate (B)node[above]{Hùng} (6,0) coordinate (C)++(-35:2.5) coordinate (A)node[below left]{Minh} ($(B)!(A)!(C)$) coordinate (H);
			\draw (B)--(A)--(H) (A)--(C) ($(B)!-.1!(H)$)--($(H)!-.1!(B)$);
			\foreach \x/\g in {A/0,B/-90,C/90,H/90}
			\fill[black] 	(\x) circle (1pt)
			($(\g:3mm)+(\x)$) node {$\x$};
			\path (A)--(B)node[pos=.45,below left]{$200$ m}
			(A)--(H)node[pos=.5,right]{$50$ m};		
			\draw pic[draw,angle radius=2mm]{right angle=B--H--A};
			\draw[-stealth] (B)++(30:.5)--++(0:1)
			;
			\draw[-stealth] (A)++(120:.3)--++(145:.6)
			;
		\end{tikzpicture}
	\end{center}
	\loigiai{
		Vận tốc của bạn Minh: $v_{1}=5$ km/h,
		vận tốc của bạn Hùng: $v_{2}=15$ km/h. \\
		Áp dụng định lý Pythagore vào tam giác vuông $AHB$, ta có 
		\allowdisplaybreaks
		\begin{eqnarray*}
			BH=\sqrt{(0{,}2)^{2}-(0{,}05)^{2}}=\dfrac{\sqrt{15}}{20} \text{ km}.
		\end{eqnarray*}
		Gọi $B C=x$ (km), $x>0$.\\
		Suy ra $C H=\dfrac{\sqrt{15}}{20}-x$, $x \leq \dfrac{\sqrt{15}}{20}$. \\
		Ta cần xác định vị trí điểm $C$ để Minh và Hùng gặp nhau mà không bạn nào phải chờ người kia nghĩa là ta cần tìm $x$ để thời gian hai bạn di chuyển đến $C$ là bằng nhau. \\
		Thời gian Hùng đi từ $B$ đến $C$ là $t_{2}=\dfrac{S_{B C}}{v_{2}}=\dfrac{x}{15}$ (h). \\
		Quãng đường $A C$ Minh đã đi là $A C=\sqrt{C H^{2}+A H^{2}}=\sqrt{\left(\dfrac{\sqrt{15}}{20}-x\right)^{2}+(0{,}05)^{2}}$. \\
		Thời gian Minh đã đi từ $A$ đến $C$ là $t_{1}=\dfrac{S_{A C}}{v_{1}}=\dfrac{\sqrt{\left(\dfrac{\sqrt{15}}{20}-x\right)^{2}+(0{,}05)^{2}}}{5}$ (h). \\
		Theo yêu cầu bài toán $\dfrac{\sqrt{\left(\dfrac{\sqrt{15}}{20}-x\right)^{2}+(0{,}05)^{2}}}{5}=\dfrac{x}{15}$. \\
		Bình phương 2 vế:
		\allowdisplaybreaks
		\begin{eqnarray*}
			&&\dfrac{\left(\dfrac{\sqrt{15}}{20}-x\right)^2+(0{,}05)^2}{25}=\dfrac{x^2}{225}\\
			&\Leftrightarrow& 9\left(\dfrac{3}{80}-\dfrac{\sqrt{15}}{10} x+x^2\right)+\dfrac{9}{400}=x^2	 \\
			&\Leftrightarrow& 8x^2-\dfrac{9\sqrt{15}}{10} x+\dfrac{9}{25}=0\\
			&\Leftrightarrow&
			\hoac{&x \approx 0{,}3\\
				&x \approx 0{,}1.}
		\end{eqnarray*}
		Vì $0<x \leq \dfrac{\sqrt{15}}{20} \approx 0{,}19$ nên $x \approx 0{,}1$ thỏa mãn. \\
		Vậy hai bạn Minh và Hùng di chuyển đến vị trí $C$ cách $B$ một đoạn $x \approx 0{,}1$ km $=100$ m.
	}

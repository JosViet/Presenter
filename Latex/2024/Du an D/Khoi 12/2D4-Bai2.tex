\subsubsection{Diện tích hình thang cong}
\indam{Định nghĩa:}
\begin{boxdn}
		Cho hàm số $y=f(x)$ liên tục và không âm trên đoạn $[a;b]$. Hình phẳng giới hạn bởi đồ thị hàm số $y=f(x)$, trục hoành và hai đường thẳng $x=a$, $x=b$ được gọi là \textit{hình thang cong}.
	\immini{Ta sẽ đưa ra công thức tính diện tích $S$ của hình thang cong này.\\
		Kí hiệu $S(x)$ là diện tích hình thang cong giới hạn bởi đồ thị hàm số $y=f(x)$, trục hoành và hai đường thẳng vuông góc với trục $Ox$ tại các điểm có hoành độ là $a$ và $x$. Ta được hàm số $S(x)$ xác định trên đoạn $[a;b]$.\\
		Người ta chứng minh được rằng: $S(x)$ là một nguyên hàm của hàm số $f(x)$ trên đoạn $[a;b]$.

	Giả sử $F(x)$ là một nguyên hàm của $f(x)$ trên $[a;b]$. Khi đó, tồn tại hằng số $C$ sao cho $S(x)=F(x)+C$. Vì $S(a)=0$ nên $F(a)+C=0$ hay $C=-F(a)$.\\
	Từ đó, $S=S(b)+C=F(b)-F(a)$.\\
	Vậy ta có kết quả sau:
	\begin{tcolorbox}[colback=orange!10!white,breakable]
		Nếu hàm số $y=f(x)$ liên tục và không âm trên đoạn $[a;b]$ thì diện tích $S$ của hình thang cong giới hạn bởi đồ thị hàm số $y=f(x)$, trục hoành và hai đường thẳng $x=a$, $x=b$ được tính bởi
		\[
		S=F(b)-F(a),
		\]
		trong đó $F(x)$ là một nguyên hàm của $f(x)$ trên đoạn $[a;b]$.
	\end{tcolorbox}
\end{boxdn}
\begin{nx}
	Cho $f(x)$ là hàm số liên tục trên đoạn $[a;b]$. Giả sử $F(x)$ và $G(x)$ là hai nguyên hàm của $f(x)$ trên đoạn $[a;b]$. Khi đó, $G(x)=F(x)+C$ với hằng số $C$ nào đó. Do đó
	\[
	G(b)-G(a)=F(b)+C-\left[F(a)+C\right]=F(b)-F(a).
	\]
	Như vậy, hiệu số $F(b)-F(a)$ không phụ thuộc vào việc chọn nguyên hàm $F(x)$ của $f(x)$.	
\end{nx}
\subsubsection{Khái niệm tích phân}
\begin{boxdn}
		Cho hàm số $f(x)$ liên tục trên đoạn $[a;b]$. Nếu $F(x)$ là một nguyên hàm của $f(x)$ trên đoạn $[a;b]$ thì hiệu số $F(b)-F(a)$ gọi là \textbf{tích phân} từ $a$ đến $b$ của hàm số $f(x)$, kí hiệu $\displaystyle\int\limits_{a}^{b}f(x)\,\mathrm{d}x$.
\end{boxdn}
	Hiệu số $F(b)-F(a)$ còn được kí hiệu là $F(x)\bigg|_{a}^{b}$.\\
	Vậy $\displaystyle\int\limits_{a}^{b}f(x)\,\mathrm{d}x=F(x)\bigg|_{a}^{b}=F(b)-F(a)$.\\
	Ta gọi $\displaystyle\int\limits_{a}^{b}$ là dấu tích phân, $a$ và $b$ là cận tích phân, $a$ là cận dưới, $b$ là cận trên, $f(x)\,\mathrm{d}x$ là biểu thức dưới dấu tích phân và $f(x)$ là hàm số dưới dấu tích phân.
\begin{note}
	\begin{enumerate}
		\item Trong trường hợp $a=b$ hoặc $a>b$, ta quy ước $\displaystyle\int\limits_{a}^{a}f(x)\,\mathrm{d}x=0$ và $\displaystyle\int\limits_{a}^{b}f(x)\,\mathrm{d}x=-\int\limits_{b}^{a}f(x)\,\mathrm{d}x$.
		\item Người ta chứng minh được rằng, tích phân chỉ phụ thuộc vào hàm số $f$ và các cận $a$, $b$ mà không phụ thuộc vào biến số $x$ hay $t$, nghĩa là $\displaystyle\int\limits_{a}^{b}f(x)\,\mathrm{d}x=\int\limits_{a}^{b}f(t)\,\mathrm{d}t$.
		\item Ý nghĩa hình học của tích phân.\\
		\immini{Nếu hàm số $y=f(x)$ liên tục và không âm trên đoạn $[a;b]$ thì $\displaystyle\int\limits_{a}^{b}f(x)\,\mathrm{d}x$ là diện tích $S$ của hình thang cong giới hạn bởi đồ thị hàm số $f(x)$, trục hoành và hai đường thẳng $x=a$, $x=b$.\\
			Vậy $S=\displaystyle\int\limits_{a}^{b}f(x)\,\mathrm{d}x$.}{\begin{tikzpicture}[scale=1, font=\footnotesize, line join=round, line cap=round,>=stealth]
				\def \xmin{-1.0};
				\def \xmax{5.7};
				\def \ymin{-1.0};

				{plot[domain=1:4.5] (\x,{-(1/10)*((\x)-0.5)*((\x)-2)*((\x)-5)+2})--(4.5,2.5)--(4.5,0)}--(1,0)--cycle;
				\draw (1,0)--(1,1.8) (4.5,0)--(4.5,2.5);

	\end{enumerate}
\end{note}
\begin{note}
	\begin{enumerate}
		\item Nếu hàm số $f(x)$ có đạo hàm $f'(x)$ và $f'(x)$ liên tục trên đoạn $[a; b]$ thì
		\[f(b)-f(a)=\displaystyle\int\limits_a^b f'(x) \,\mathrm{d}x.\]
		\item Ta đã biết rằng, đạo hàm của quãng đường di chuyển theo thời gian bằng tốc độ của chuyển động tại mỗi thời điểm $\left(v(t)=s'(t)\right)$. Do đó, nếu biết tốc độ $v(t)$ tại mọi thời điểm $t \in[a; b]$ thì tính được quãng đường di chuyển trong khoảng thời gian từ $a$ đến $b$ theo công thức
		\[s=s(b)-s(a)=\displaystyle\int\limits_a^b v(t) \,\mathrm{d}t.\]
	\end{enumerate}
\end{note}
\begin{nx}
	Cho hàm số $f(x)$ liên tục trên đoạn $[a; b]$. Khi đó $\dfrac{1}{b-a} \displaystyle\int\limits_a^b f(x) \,\mathrm{d} x$ được gọi là giá trị trung bình của hàm số $f(x)$ trên đoạn $[a; b]$.
\end{nx}

\begin{dang}{Tính diện tích hình thang cong}
	{\bf{Phương pháp giải}}
	\begin{itemize}
		\item \textit{Bước 1.}  Kiểm tra điều kiện: Đảm bảo hàm số $y=f(x)$ liên tục và không âm trên đoạn $[a;b]$. Nếu hàm số có giá trị âm hoặc đổi dấu trên đoạn $[a;b]$, cần chia khoảng và lấy trị tuyệt đối của hàm số (nội dung này sẽ được học kỹ hơn ở các bài sau, trong phạm vi bài này, các hàm số thường là không âm).
		\item \textit{Bước 2.} Tìm nguyên hàm: Tìm một nguyên hàm $F(x)$ của hàm số $f(x)$.
		\item \textit{Bước 3.} Áp dụng công thức: Diện tích $S$ được tính bằng công thức: $S=F(b)-F(a)=\displaystyle\int\limits_a^b f(x)\,\mathrm{d}x$.
	\end{itemize}
\end{dang}

\begin{ex}%[2D4V2-2]
	[Trích đề thi GHKII - THPT Thanh Miện - Hải Dương - Năm học 2024-2025] 
	Cho hàm số $f(x) = \heva{&
		2x^2 - 1 &&\text{ khi } x < 0 \\

		&5 - 2x &&\text{ khi } x > 2
	}$ và $I = \displaystyle\int\limits\limits_{-5}^{9} f(x) \,\mathrm{d}x$. \\

		Ta có
		\allowdisplaybreaks
		\begin{eqnarray*}
			I &=& \displaystyle\int\limits_{-5}^{9} f(x) \,\mathrm{d}x \\
			&=& \displaystyle\int\limits_{-5}^{0} (2x^2 - 1) \,\mathrm{d}x + \displaystyle\int\limits_{0}^{2} (x - 1) \,\mathrm{d}x + \displaystyle\int\limits_{2}^{9} (5 - 2x) \,\mathrm{d}x \\
			&=& \left( \dfrac{2x^3}{3} - x \right) \Bigg|_{-5}^{0} + \left( \dfrac{x^2}{2} - x \right) \Bigg|_{0}^{2} + \left( 5x - x^2 \right) \Bigg|_{2}^{9} \\

			&=& \dfrac{109}{3}.
		\end{eqnarray*}
		Suy ra $a = 109$, $b = 3$.\\
		Do đó $a - b = 109 - 3 = 106$.
	}
